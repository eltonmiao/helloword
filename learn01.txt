//按钮控件 
.btn {
  display: inline-block;
  padding: 6px 12px;
  margin-bottom: 0;
  font-size: 14px;
  font-weight: normal;
  line-height: 1.42857143;
  text-align: center;
  white-space: nowrap; // 规定段落中的文本不进行换行
  vertical-align: middle;
  -ms-touch-action: manipulation;
      touch-action: manipulation;
  cursor: pointer; // 手型图标
  -webkit-user-select: none;
     -moz-user-select: none;
      -ms-user-select: none;
          user-select: none;
  background-image: none;
  border: 1px solid transparent; // 不显示默认边框
  border-radius: 4px;
}
// 按钮的状态 包括 focus（选中） hover（悬停） active（点击）
// 这些状态可以应用到 button input a 标签上
<input class="btn btn-default" type="button" value="Input">
<a class="btn btn-default" href="#123">Link</a>
<input class="form-control input-lg" type="text" placeholder=".input-lg"> 

.btn:active, // 点击起作用的伪类
.btn.active { // 手动添加到标签上的类
  background-image: none;
  outline: 0;
  -webkit-box-shadow: inset 0 3px 5px rgba(0, 0, 0, .125); // 点击时，的阴影效果
          box-shadow: inset 0 3px 5px rgba(0, 0, 0, .125);
}
// 按钮即被选中又处于点击时的状态。 相当于多个状态同时发生
.btn-default:active:hover,
.btn-default.active:hover,
.open > .dropdown-toggle.btn-default:hover,
.btn-default:active:focus, // 已经被选中的元素又被点击时的状态
.btn-default.active:focus,
.open > .dropdown-toggle.btn-default:focus,
.btn-default:active.focus,
.btn-default.active.focus,
.open > .dropdown-toggle.btn-default.focus {
  color: #333;
  background-color: #d4d4d4;
  border-color: #8c8c8c;
}


// 按钮处于disable状态下
.btn.disabled,
.btn[disabled],
fieldset[disabled] .btn {
  cursor: not-allowed;
  filter: alpha(opacity=65);
  -webkit-box-shadow: none;
          box-shadow: none;
  opacity: .65;
}
